update parcelas pc set objetos = (select genera_objetos(obj1, obj2, obj3, obj4, obj5, obj6, obj7, obj8, obj9) 
					from planospos2000 pl
					where pc.plano = pl.plano
					limit 1);
					
update parcelas pr set objetos = (select genera_objetos(obj1,obj2,obj3,obj4,obj5,obj6,obj7,obj8,obj9) 
				from planosant2000 pl
				where pl.obj1 != '' and pr.plano = pl.plano
				limit 1);
